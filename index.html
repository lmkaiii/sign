        let currentStep = 1;

            }

        // ==================== 電子簽名功能 ====================

        let canvas, ctx, isDrawing = false, lastX = 0, lastY = 0;

        let signatureHistory = [];


        function initSignaturePad() {

            canvas = document.getElementById('signature-canvas');

            ctx = canvas.getContext('2d');

            ctx.lineWidth = 3;

            ctx.lineCap = 'round';

            ctx.strokeStyle = '#1e40af';


            // 滑鼠事件

            canvas.addEventListener('mousedown', startDrawing);

            canvas.addEventListener('mousemove', draw);

            canvas.addEventListener('mouseup', stopDrawing);

            canvas.addEventListener('mouseout', stopDrawing);


            // 觸控事件 (手機/平板)

            canvas.addEventListener('touchstart', handleTouchStart);

            canvas.addEventListener('touchmove', handleTouchMove);

            canvas.addEventListener('touchend', stopDrawing);

        }


        function startDrawing(e) {

            isDrawing = true;

            const rect = canvas.getBoundingClientRect();

            lastX = e.clientX - rect.left;

            lastY = e.clientY - rect.top;

            signatureHistory.push(ctx.getImageData(0, 0, canvas.width, canvas.height));

        }


        function draw(e) {

            if (!isDrawing) return;

            const rect = canvas.getBoundingClientRect();

            const x = e.clientX - rect.left;

            const y = e.clientY - rect.top;


            ctx.beginPath();

            ctx.moveTo(lastX, lastY);

            ctx.lineTo(x, y);

            ctx.stroke();


            lastX = x;

            lastY = y;

        }


        function stopDrawing() {

            if (isDrawing) {

                isDrawing = false;

                signatureHistory.push(ctx.getImageData(0, 0, canvas.width, canvas.height));

            }

        }


        function handleTouchStart(e) {

            e.preventDefault();

            isDrawing = true;

            const rect = canvas.getBoundingClientRect();

            const touch = e.touches[0];

            lastX = touch.clientX - rect.left;

            lastY = touch.clientY - rect.top;

            signatureHistory.push(ctx.getImageData(0, 0, canvas.width, canvas.height));

        }


        function handleTouchMove(e) {

            e.preventDefault();

            if (!isDrawing) return;

            const rect = canvas.getBoundingClientRect();

            const touch = e.touches[0];

            const x = touch.clientX - rect.left;

            const y = touch.clientY - rect.top;


            ctx.beginPath();

            ctx.moveTo(lastX, lastY);

            ctx.lineTo(x, y);

            ctx.stroke();


            lastX = x;

            lastY = y;

        }


        function clearSignature() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            signatureHistory = [];

        }


        function undoSignature() {

            if (signatureHistory.length > 1) {

                signatureHistory.pop();

                ctx.putImageData(signatureHistory[signatureHistory.length - 1], 0, 0);

            } else if (signatureHistory.length === 1) {

                ctx.clearRect(0, 0, canvas.width, canvas.height);

                signatureHistory = [];

            }

        }


        // ==================== PDF 生成 ====================

        async function generatePDF() {

            const nameInput = document.getElementById('english-name').value.trim();

            const dateInput = document.getElementById('sign-date').value;


            if (!nameInput) {

                alert('請填寫英文姓名');

                return;

            }

            if (!ctx || ctx.getImageData(0,0,canvas.width,canvas.height).data.every(pixel => pixel === 0)) {

                alert('請繪寫電子簽名');

                return;

            }


            const container = document.getElementById('form-container');

            

            // 使用 html2canvas 捕獲完整視覺 (包含圖片 + 填寫內容 + 簽名)

            const canvasCapture = await html2canvas(container, {

                scale: 2,           // 高解析度

                useCORS: true,

                backgroundColor: '#ffffff'

            });


            const imgData = canvasCapture.toDataURL('image/png');


            // 生成 PDF (A4 直向)

            const { jsPDF } = window.jspdf;

            const pdf = new jsPDF({

                orientation: 'portrait',

                unit: 'mm',

                format: 'a4'

            });


            // 計算圖片比例以適配 A4

            const pdfWidth = 210;

            const pdfHeight = 297;

            const imgWidth = canvasCapture.width;

            const imgHeight = canvasCapture.height;

            const ratio = Math.min(pdfWidth / imgWidth, pdfHeight / imgHeight);

            const finalWidth = imgWidth * ratio;

            const finalHeight = imgHeight * ratio;

            const xOffset = (pdfWidth - finalWidth) / 2;

            const yOffset = (pdfHeight - finalHeight) / 2;


            pdf.addImage(imgData, 'PNG', xOffset, yOffset, finalWidth, finalHeight);


            // 檔名

            const fileName = `BGCA_確認表_${nameInput.replace(/[^a-zA-Z0-9]/g, '_')}_${dateInput}.pdf`;

            

            pdf.save(fileName);

            

            alert(`✅ PDF 已成功生成並下載！\n檔名：${fileName}\n\n您已完成確認程序。`);

        }


        // 設定當天日期 (香港格式 DD/MM/YYYY)

        function setTodayDate() {

            const today = new Date();

            const day = String(today.getDate()).padStart(2, '0');

            const month = String(today.getMonth() + 1).padStart(2, '0');

            const year = today.getFullYear();

            document.getElementById('sign-date').value = `${day}/${month}/${year}`;

        }
